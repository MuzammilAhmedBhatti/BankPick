package com.example.smd_6a_2;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

public class MainActivity extends AppCompatActivity {
    // hooks
    EditText etID;
    Button btnCancel, btnSubmit;
    TextView tvResult;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        init();

        btnCancel.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                finish();
            }
        });

        btnSubmit.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                String id = etID.getText().toString().trim();

                if(id.length()!=13)
                {
                    //etID.setError("ID length must be equal to 13");
                    Toast.makeText(MainActivity.this, "ID length must be equal to 13", Toast.LENGTH_SHORT).show();
                    return;
                }

                String rawDob = id.substring(0, 6);

                String year = "20"+rawDob.substring(0, 2); // 2019
                int day = Integer.parseInt(rawDob.substring(4, 6));
                String []months = {"", "January", "February", "March", "April"};
                String dob = day+"/"+months[Integer.parseInt(rawDob.substring(2, 4))]+"/"+year;
                String gender;

                if(Integer.parseInt(String.valueOf(id.charAt(6)))<5)
                {
                    gender = "Female";
                }
                else
                {
                    gender = "Male";
                }

                StringBuilder sb = new StringBuilder();
                sb.append("Date of Birth : ").append(dob).append("\n").append("Gender : ").append(gender);
                tvResult.setText(sb.toString());

            }
        });

    }

    private void init()
    {
        etID = findViewById(R.id.etID);
        btnCancel = findViewById(R.id.btnCancel);
        btnSubmit = findViewById(R.id.btnSubmit);
        tvResult = findViewById(R.id.tvResult);
    }
}

public class MainActivity extends AppCompatActivity {

    Button btnOpenSecondActiivty, btnGetInfo;
    TextInputEditText tietName;
    TextView tvSecondName;
    ImageView ivCall, ivMap, ivBrowser;

    ActivityResultLauncher<Intent> getInfoLauncher;
    String name, url, phone, address;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });
        init();

        getInfoLauncher = registerForActivityResult(
                new ActivityResultContracts.StartActivityForResult(),
                result->{
                    if(result.getResultCode() == RESULT_CANCELED)
                    {
                        Toast.makeText(this, "You have to give the information", Toast.LENGTH_SHORT).show();
                    }
                    else if(result.getResultCode()==RESULT_OK && result.getData()!=null)
                    {
                        Intent i = result.getData();
                        name = i.getStringExtra("name_key");
                        url = i.getStringExtra("url_key");
                        address = i.getStringExtra("address_key");
                        phone = i.getStringExtra("phone_key");

                        ivBrowser.setVisibility(VISIBLE);
                        tvSecondName.setVisibility(VISIBLE);
                        ivCall.setVisibility(VISIBLE);
                        ivMap.setVisibility(VISIBLE);
                        tvSecondName.setText(name);
                    }
                }
        );

        btnOpenSecondActiivty.setOnClickListener((view)->{
            String name = tietName.getText().toString().trim();
            if(name.length()==0)
            {
                tietName.setError("Name can't be empty.");
                return;
            }

            Intent intent = new Intent(MainActivity.this, SecondActivity.class);
            intent.putExtra("name_key", name);
            startActivity(intent);
            finish();
        });

        btnGetInfo.setOnClickListener((v)->{
            getInfoLauncher.launch(new Intent(MainActivity.this, GetInfoActivity.class));
        });

        ivBrowser.setOnClickListener((v)->{
            Intent i = new Intent(Intent.ACTION_VIEW);
            i.setData(Uri.parse(url));
            startActivity(i);
        });

        ivCall.setOnClickListener((v)->{
            Intent i = new Intent(Intent.ACTION_DIAL);
            i.setData(Uri.parse("tel:"+phone));
            startActivity(i);
        });

        ivMap.setOnClickListener((v)->{
            Intent i = new Intent(Intent.ACTION_VIEW);
            i.setData(Uri.parse("google.navigation:q="+address));
            i.setPackage("com.google.android.apps.maps");
            startActivity(i);
        });
    }

    private void init()
    {
        btnOpenSecondActiivty = findViewById(R.id.btnOpenSecondActiivty);
        tietName = findViewById(R.id.tietName);
        tvSecondName = findViewById(R.id.tvSecondName);
        ivBrowser = findViewById(R.id.ivBrowser);
        ivCall = findViewById(R.id.ivCall);
        ivMap = findViewById(R.id.ivMap);
        btnGetInfo = findViewById(R.id.btnGetInfo);

        tvSecondName.setVisibility(GONE);
        ivMap.setVisibility(GONE);
        ivCall.setVisibility(GONE);
        ivBrowser.setVisibility(GONE);
    }

}

public class Dashboard extends AppCompatActivity implements ListFrag.OnItemClicked {

    ArrayList<String> items;

    View view, isPortrait;

    DetailFrag detailFrag ;

    FragmentManager manager;



    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_dashboard);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        manager = getSupportFragmentManager();
        detailFrag = (DetailFrag) manager.findFragmentById(R.id.detail_frag);
        view = detailFrag.requireView();


        tvDetail = view.findViewById(R.id.tvDetail);
        isPortrait = findViewById(R.id.portrait);
        if (isPortrait!=null) {
            // hide detail fragment and show list fragment
            manager.beginTransaction()
                    .hide(detailFrag)
                    .show(manager.findFragmentById(R.id.list_frag))
                    .commit();
        }
        else
        {
            // show detail fragment and show list fragment
            manager.beginTransaction()
                    .show(detailFrag)
                    .show(manager.findFragmentById(R.id.list_frag))
                    .commit();
        }

        items = new ArrayList<>();
        items.add("Description Item 1");
        items.add("Description Item 2");
        items.add("Description Item 3");
        items.add("Description Item 4");
        items.add("Description Item 5");
        items.add("Description Item 6");
        items.add("Description Item 7");


    }

    @Override
    public void onClick(int index) {
        tvDetail.setText(items.get(index));
        if(isPortrait != null)
        {
            manager.beginTransaction()
                    .show(detailFrag)
                    .hide(manager.findFragmentById(R.id.list_frag))
                    .addToBackStack(null)
                    .commit();
        }
    }
}

public class CourseAdapter extends ArrayAdapter<Course> {
    Context context;
    Course selectedCourse;
    View dialogView;
    AlertDialog dialog;
    public CourseAdapter(@NonNull Context context, @NonNull List<Course> objects) {
        super(context, 0, objects);
        this.context = context;
    }

    @NonNull
    @Override
    public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {

            convertView = LayoutInflater.from(context)
                    .inflate(R.layout.course_single_item_design, parent, false);

        Course course = getItem(position);

        TextView tvCC = convertView.findViewById(R.id.tvCC);
        TextView tvCH = convertView.findViewById(R.id.tvCH);
        TextView tvCN = convertView.findViewById(R.id.tvCN);
        ImageView ivDel = convertView.findViewById(R.id.ivDel);
        ImageView ivEdit = convertView.findViewById(R.id.ivEdit);

        tvCC.setText(course.getCode());
        tvCN.setText(course.getName());
        tvCH.setText(String.valueOf(course.getCreditHours()));

        ivDel.setOnClickListener((v)->{
            remove(course);
            Toast.makeText(context, "Course deleted succesfully", Toast.LENGTH_SHORT).show();
        });

        ivEdit.setOnClickListener((v)->{
            editCourse(course);
        });


        return convertView;
    }

    private void editCourse(Course course) {
        selectedCourse = course;
        if (dialog == null) {
            dialogView = LayoutInflater.from(context)
                    .inflate(R.layout.add_update_course_dialog_design, null);
            dialog = new AlertDialog.Builder(context)
                    .setTitle("Edit Course")
                    .setView(dialogView)
                    .create();
            Button btnCancel = dialogView.findViewById(R.id.btnDialogCancel);
            Button btnEdit = dialogView.findViewById(R.id.btnDialogAdd);
            btnEdit.setText("Edit");

            EditText etCC = dialogView.findViewById(R.id.etCC);
            EditText etCH = dialogView.findViewById(R.id.etCH);
            EditText etCN = dialogView.findViewById(R.id.etCN);

            btnEdit.setOnClickListener(v1->{
                selectedCourse.setCode(etCC.getText().toString());
                selectedCourse.setName(etCN.getText().toString());
                selectedCourse.setCreditHours(Integer.parseInt(etCH.getText().toString()));
                notifyDataSetChanged();
                dialog.dismiss();
            });

            btnCancel.setOnClickListener(v1->{
                dialog.dismiss();
            });
        }
        EditText etCC = dialogView.findViewById(R.id.etCC);
        EditText etCH = dialogView.findViewById(R.id.etCH);
        EditText etCN = dialogView.findViewById(R.id.etCN);

        etCH.setText(String.valueOf(course.getCreditHours()));
        etCC.setText(course.getCode());
        etCN.setText(course.getName());

        dialog.show();
    }

    @Override
    public int getCount() {
        return super.getCount();
    }
}

                editor.putString("key_username", username);
                editor.putString("key_password", pass);
                editor.putBoolean("isLogin", true);
                editor.commit();
                Toast.makeText(this, "Registered", Toast.LENGTH_SHORT).show();
                startActivity(new Intent(Singup.this, MainActivity.class));
                finish();

                return;
            }

            Toast.makeText(this, "Password mis-match", Toast.LENGTH_SHORT).show();

        });

    }

        btnCancel = findViewById(R.id.btnCancel);
        btnSignup = findViewById(R.id.btnSignup);
        tvLogin = findViewById(R.id.tvLogin);
        etPassword = findViewById(R.id.etPassword);
        etConfirmPassword = findViewById(R.id.etConfirmPassword);
        etUsername = findViewById(R.id.etUsername);
        sPref = getSharedPreferences("user", MODE_PRIVATE);
        editor = sPref.edit();
    }

package com.example.smd_6a_s26_8;

import android.app.AlertDialog;
import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ImageView;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.recyclerview.widget.RecyclerView;

import com.google.android.material.textfield.TextInputEditText;

import java.util.ArrayList;

public class PassengerAdapter extends RecyclerView.Adapter<PassengerAdapter.PassengerViewHolder> {

    Context context;
    ArrayList<Passenger> passengers;
    public PassengerAdapter(Context context, ArrayList<Passenger> passengers)
    {
        this.context = context;
        this.passengers = passengers;
    }


    @NonNull
    @Override
    public PassengerViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        View view = LayoutInflater.from(context)
                .inflate(R.layout.single_passenger_item_design, parent, false);
        return new PassengerViewHolder(view);
    }

    @Override
    public void onBindViewHolder(@NonNull PassengerViewHolder holder, int position) {
        Passenger passenger = passengers.get(position);
        holder.tvName.setText(passenger.getName());
        holder.tvPhone.setText(passenger.getPhone());
        String preference = passenger.getPreference();
        if(preference.equals("bus"))
        {
            holder.ivPref.setImageDrawable(context.getDrawable(R.drawable.icon_bus));
        }
        else if(preference.equals("train"))
        {
            holder.ivPref.setImageDrawable(context.getDrawable(R.drawable.icon_train));
        }
        else if(preference.equals("plane"))
        {
            holder.ivPref.setImageDrawable(context.getDrawable(R.drawable.icon_plane));
        }


        holder.itemView.setOnClickListener((v)->{
            AlertDialog.Builder confirmationBuilder = new AlertDialog.Builder(context)
                    .setTitle("Confirmation")
                    .setMessage("What you want?")
                    .setPositiveButton("Update", (a, b)->{
                        updatePassenger(passenger, position);
                    })
                    .setNegativeButton("Delete", (a,b)->{
                        passengers.remove(passenger);
                        notifyItemRemoved(position);
                    });

            confirmationBuilder.create().show();
        });
    }

    private void updatePassenger(Passenger passenger, int position) {

        View view
                = LayoutInflater.from(context)
                .inflate(R.layout.add_update_passenger_dialog, null);

        TextInputEditText etName = view.findViewById(R.id.etName);
        TextInputEditText etPhone = view.findViewById(R.id.etPhone);
        TextInputEditText etPref = view.findViewById(R.id.etPref);

        etName.setText(passenger.getName());
        etPhone.setText(passenger.getPhone());
        etPref.setText(passenger.getPreference());

        androidx.appcompat.app.AlertDialog.Builder builder =
                new androidx.appcompat.app.AlertDialog.Builder(context)
                .setTitle("Update Passenger")
                .setView(view)
                .setPositiveButton("Save", (a,b)->{

                                   passenger.setName(etName.getText().toString().trim());
                                    passenger.setPhone(etPhone.getText().toString().trim());
                                    passenger.setPreference(etPref.getText().toString().trim());
                                    notifyItemChanged(position);
                })
                .setNegativeButton("Cancel", null);

        builder.create().show();
    }

    @Override
    public int getItemCount() {
        return passengers.size();
    }

    public class PassengerViewHolder extends RecyclerView.ViewHolder
    {
        ImageView ivPref;
        TextView tvName;
        TextView tvPhone;

        public PassengerViewHolder(@NonNull View itemView) {
            super(itemView);
            ivPref = itemView.findViewById(R.id.ivPref);
            tvName = itemView.findViewById(R.id.tvName);
            tvPhone = itemView.findViewById(R.id.tvPhone);
        }
    }
}

package com.example.smd_6a_s26_10;

import android.content.ContentValues;
import android.content.Context;
import android.database.Cursor;
import android.database.sqlite.SQLiteDatabase;
import android.database.sqlite.SQLiteOpenHelper;

import androidx.annotation.Nullable;

import java.util.ArrayList;

public class DatabaseManager {
    static final String DATABASE_NAME = "LibraryDB";
    static final int DATABASE_VERSION = 2;
    static final String TABLE_BOOK = "books";
    static final String COLUMN_ID = "book_id";
    static final String COLUMN_NAME = "book_name";
    static final String COLUMN_AUTHOR = "book_author";
    static final String COLUMN_DATE = "publication_date";

    Context context;
    DBHelper helper;



    public DatabaseManager(Context context)
    {
        this.context = context;
    }

    public void open()
    {
        helper = new DBHelper(context);
    }

    public long insertBook(Book book)
    {
        SQLiteDatabase db = helper.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put(COLUMN_NAME, book.getBookName());
        cv.put(COLUMN_AUTHOR, book.getBookAuthor());
        cv.put(COLUMN_DATE, book.getPublicationDate());
        long id = db.insert(TABLE_BOOK, null, cv);
        db.close();
        return id;
    }

    public Book getBookById(int id)
    {
        SQLiteDatabase db = helper.getReadableDatabase();

        Cursor cursor = db.query(TABLE_BOOK,
                null,
                COLUMN_ID+"=?",
                new String[]{String.valueOf(id)},
                null, null, null);

        Book book = null;

        if(cursor!=null && cursor.moveToFirst())
        {
            int index_name = cursor.getColumnIndex(COLUMN_NAME);
            int index_author = cursor.getColumnIndex(COLUMN_AUTHOR);
            int index_date = cursor.getColumnIndex(COLUMN_DATE);

            book = new Book();
            book.setBookId(id);
            book.setBookName(cursor.getString(index_name));
            book.setBookAuthor(cursor.getString(index_author));
            book.setPublicationDate(cursor.getString(index_date));
            cursor.close();
        }
        db.close();
        return book;
    }

    public ArrayList<Book> getAllBooks()
    {
        SQLiteDatabase db = helper.getReadableDatabase();

        Cursor cursor = db.query(TABLE_BOOK,
                null,
                null,
                null,
                null, null, null);

        ArrayList<Book> books = new ArrayList<>();

        if(cursor!=null && cursor.moveToFirst())
        {
            do {
                int index_name = cursor.getColumnIndex(COLUMN_NAME);
                int index_id = cursor.getColumnIndex(COLUMN_ID);
                int index_author = cursor.getColumnIndex(COLUMN_AUTHOR);
                int index_date = cursor.getColumnIndex(COLUMN_DATE);

                Book book = new Book();
                book.setBookId(cursor.getInt(index_id));
                book.setBookName(cursor.getString(index_name));
                book.setBookAuthor(cursor.getString(index_author));
                book.setPublicationDate(cursor.getString(index_date));
                books.add(book);
            }while(cursor.moveToNext());

            cursor.close();
        }

        db.close();
        return books;
    }

    public int update(Book book)
    {
        SQLiteDatabase db = helper.getWritableDatabase();
        ContentValues cv = new ContentValues();
        cv.put(COLUMN_NAME, book.getBookName());
        cv.put(COLUMN_AUTHOR, book.getBookAuthor());
        cv.put(COLUMN_DATE, book.getPublicationDate());
        int id = db.update(TABLE_BOOK,
                cv,
                COLUMN_ID+"=?",
                new String[]{String.valueOf(book.getBookId())});
        db.close();
        return id;
    }

    public int deleteBook(int id)
    {
        SQLiteDatabase db = helper.getWritableDatabase();
        int count = db.delete(TABLE_BOOK, COLUMN_ID+"=?",
                new String[]{String.valueOf(id)});
        db.close();
        return count;
    }

    public void close()
    {
        helper.close();
    }


    private class DBHelper extends SQLiteOpenHelper
    {

        public DBHelper(@Nullable Context context) {
            super(context, DATABASE_NAME, null, DATABASE_VERSION);
        }

        @Override
        public void onCreate(SQLiteDatabase sqLiteDatabase) {
            // schema
            String createTable = "CREATE TABLE "+TABLE_BOOK+"("
                    +COLUMN_ID+" INTEGER PRIMARY KEY AUTOINCREMENT, "
                    +COLUMN_NAME+" TEXT, "
                    +COLUMN_AUTHOR+" TEXT, "
                    +COLUMN_DATE+" TEXT)";
            sqLiteDatabase.execSQL(createTable);
        }

        @Override
        public void onUpgrade(SQLiteDatabase sqLiteDatabase, int i, int i1) {
            // backup data
            // drop
            sqLiteDatabase.execSQL("DROP TABLE IF EXISTS "+TABLE_BOOK);
            onCreate(sqLiteDatabase);
        }
    }

}
